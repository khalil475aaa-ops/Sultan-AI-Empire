    <script>
        let recognition;
        let isListening = false;

        // --- 1. Patient Mic (6s Wait) - Kept as per policy ---
        function toggleVoice() {
            const micBtn = document.getElementById('startMic');
            if (isListening) { recognition.stop(); return; }

            recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)();
            recognition.lang = 'ur-PK'; 
            recognition.continuous = true;
            recognition.interimResults = true;

            recognition.onstart = () => { isListening = true; micBtn.classList.add('active'); };

            let silenceTimer;
            recognition.onresult = (event) => {
                clearTimeout(silenceTimer);
                for (let i = event.resultIndex; i < event.results.length; ++i) {
                    if (event.results[i].isFinal) {
                        document.getElementById('sultanInput').value += event.results[i][0].transcript + " ";
                    }
                }
                silenceTimer = setTimeout(() => { recognition.stop(); }, 6000); 
            };

            recognition.onend = () => { isListening = false; micBtn.classList.remove('active'); };
            recognition.onerror = () => { isListening = false; micBtn.classList.remove('active'); };
            recognition.start();
        }

        // --- 2. Advanced Pakistani Urdu Speaker ---
        function speakAnyText() {
            let textToSpeak = window.getSelection().toString();
            if (!textToSpeak) {
                textToSpeak = document.getElementById('terminalOutput').innerText;
            }
            if (textToSpeak.trim() === "") return;

            const utterance = new SpeechSynthesisUtterance(textToSpeak);
            
            // چیک کریں کہ کیا تحریر میں اردو حروف ہیں
            const isUrdu = /[ا-ی]/.test(textToSpeak);

            if (isUrdu) {
                utterance.lang = 'ur-PK'; // پاکستانی اردو
                utterance.pitch = 1.1;    // قدرتی آواز کے لیے پچ
                utterance.rate = 0.85;    // اردو کے سنجیدہ لہجے کے لیے تھوڑی کم رفتار
            } else {
                utterance.lang = 'en-US';
                utterance.rate = 0.9;
            }

            // تمام دستیاب آوازوں کو اسکین کریں اور بہترین اردو آواز منتخب کریں
            const voices = window.speechSynthesis.getVoices();
            const urduVoice = voices.find(v => v.lang.includes('ur') || v.lang.includes('hi')); 
            if (urduVoice) utterance.voice = urduVoice;

            window.speechSynthesis.cancel(); 
            window.speechSynthesis.speak(utterance);
        }

        // --- 3. Fusion Engine - Kept as per policy ---
        function runFusionEngine() {
            const input = document.getElementById('sultanInput').value;
            const terminal = document.getElementById('terminalOutput');
            const preview = document.getElementById('livePreview').contentWindow.document;

            if(!input) { alert("Sultan, پہلے حکم دیں!"); return; }

            terminal.innerHTML += `<br><span style="color:white">> Task: ${input}</span>`;
            terminal.innerHTML += `<br><span style="color:var(--neon-green)">> Processing through Pakistani AI Brain...</span>`;
            
            const demoCode = `<html><body style="background:#000; color:gold; text-align:center; padding-top:50px; font-family:sans-serif;"><h1>SULTAN VIP</h1><p style="color:white;">Urdu Analysis Complete.</p></body></html>`;
            
            preview.open(); preview.write(demoCode); preview.close();
            terminal.scrollTop = terminal.scrollHeight;
        }

        document.onkeydown = function(e) {
            if(e.keyCode == 123 || (e.ctrlKey && e.shiftKey && e.keyCode == 73)) return false;
        };
